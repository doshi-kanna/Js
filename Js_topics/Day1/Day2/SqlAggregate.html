SQL LEARNING MODULE - SCHOOL DATABASE

Database and Tables

Definition: A database stores related tables. Tables contain rows of data. Primary Key uniquely identifies each row. Foreign Key links tables together. 
CREATE DATABASE SCHOOL;
USE SCHOOL;

CREATE TABLE COURSE (
  CRS_ID INT PRIMARY KEY AUTO_INCREMENT,
  CRS_NAME VARCHAR(30),
  CRS_PRICE INT
);

CREATE TABLE TRAINER(
  TR_ID INT PRIMARY KEY AUTO_INCREMENT ,
  TR_NAME VARCHAR(40)
);

CREATE TABLE STUDENT (
  ST_ID INT PRIMARY KEY AUTO_INCREMENT ,
  ST_NAME VARCHAR(40),
  CRS_ID INT ,
  TR_ID INT 
);

ALTER TABLE STUDENT 
ADD CONSTRAINT FOREIGN KEY(CRS_ID) REFERENCES COURSE(CRS_ID);

Insert Data

INSERT is used to add new rows into a table. 
INSERT INTO COURSE (CRS_NAME ,CRS_PRICE) 
VALUES ('JAVA',20000),('SQL',22000),('PYTHON',19000),('C++',19000);

INSERT INTO TRAINER (TR_NAME )VALUES('BADRI'),('VENKAT');

INSERT INTO STUDENT (ST_NAME ,CRS_ID,TR_ID) 
VALUES ('SHYAM',1,1),('NITHIN',2,2),('AJAY',3,2),('NITHISH',2,2);

Inner Join

Definition: INNER JOIN returns matching rows between two tables based on common column. 
SELECT ST.ST_NAME, C.CRS_NAME, C.CRS_PRICE
FROM STUDENT ST 
JOIN COURSE C 
ON ST.CRS_ID = C.CRS_ID;

Left Join

Definition: LEFT JOIN returns all rows from left table and matched rows from right table. 
SELECT ST.ST_NAME, T.TR_NAME 
FROM STUDENT ST 
LEFT JOIN TRAINER T 
ON ST.TR_ID = T.TR_ID;

Right Join

Definition: RIGHT JOIN returns all rows from right table, matched rows from left table. 
SELECT ST.ST_NAME, C.CRS_NAME
FROM STUDENT ST 
RIGHT JOIN COURSE C 
ON ST.CRS_ID = C.CRS_ID;

WHERE Clause

Definition: WHERE is used to filter records based on conditions. 
SELECT * FROM COURSE WHERE CRS_PRICE >20000;

GROUP BY & Aggregate Functions

Definition: GROUP BY groups similar data, used with COUNT, SUM, AVG for aggregated results. 
SELECT CRS_ID, COUNT(ST_ID)
FROM STUDENT
GROUP BY CRS_ID;

UNION

Definition: UNION combines result of two queries and removes duplicates. 
SELECT CRS_NAME FROM COURSE 
UNION 
SELECT PROD_NAME FROM PRODUCT;

UNION ALL

Definition: UNION ALL returns all data including duplicates. 
SELECT CRS_NAME FROM COURSE 
UNION ALL
SELECT PROD_NAME FROM PRODUCT;

CROSS JOIN

Definition: CROSS JOIN returns every combination of rows from both tables. 
SELECT * FROM COURSE 
CROSS JOIN STUDENT;
